```bash
curl -X PUT "https://sgmanager.ng.bluemix.net/v1/sgconfig/<gateway_id>/destinations/<destination_id>/ipTableRule"
    -H "Authorization: Bearer <security_token>" -H "Content-type:  application/json" -d
    '{"src":"<app_IP>","app":"<app_ID>"}'  -k
```
